    ```rust
    pub fn parse_ytdlp_error(line: &str) -> Option<String> {
        // Check patterns in priority order (403 before generic to handle bot detection)
        if line.contains("HTTP Error 429") {
            return Some("YouTube is rate limiting requests. Wait a few minutes and try again.".into());
        }
        if line.contains("HTTP Error 403") {
            return Some("This video is not available in your region or YouTube is blocking the request.".into());
        }
        if line.contains("Video unavailable") || line.contains("This video is not available") {
            return Some("This video is unavailable or has been removed.".into());
        }
        if line.contains("Sign in to confirm") || line.contains("Private video") {
            return Some("This video is private and cannot be downloaded.".into());
        }
        if line.contains("ffmpeg not found") || line.contains("ffprobe and ffmpeg not found") {
            return Some("ffmpeg binary is missing. Reinstall the app.".into());
        }
        if line.contains("urlopen error") || line.contains("Unable to connect") {
            return Some("Network error. Check your internet connection.".into());
        }
        if line.contains("is not a valid URL") || line.contains("Unsupported URL") {
            return Some("This URL is not a recognized YouTube link.".into());
        }
        if line.contains("ERROR:") {
            return Some(format!("Download error: {}", line.trim()));
        }
        None
    }
    ```

    Implement the main `download` Tauri command:
    ```rust
    #[tauri::command]
    pub async fn download(
        app: tauri::AppHandle,
        url: String,
        save_dir: String,
        on_event: tauri::ipc::Channel<DownloadEvent>,
    ) -> Result<(), String> {
        let ffmpeg_path = resolve_sidecar_path(&app, "ffmpeg")?;

        let (mut rx, child) = app
            .shell()
            .sidecar("binaries/yt-dlp")
            .unwrap()
            .args([
                "-x",
                "--audio-format", "mp3",
                "--audio-quality", "0",
                "--embed-metadata",
                "--ffmpeg-location", &ffmpeg_path,
                "--progress-template",
                "download:PROGRESS %(progress._percent)f %(progress._speed_str)s %(progress._eta_str)s",
                "--newline",
                "-o", &format!("{}/%(title)s.%(ext)s", save_dir),
                &url,
            ])
            .spawn()
            .map_err(|e| format!("Failed to spawn yt-dlp: {}", e))?;

        // Store child handle in AppState for cleanup
        {
            let state = app.state::<crate::state::AppState>();
            let mut guard = state.active_child.lock().map_err(|e| e.to_string())?;
            *guard = Some(child);
        }

        tauri::async_runtime::spawn(async move {
            use tauri_plugin_shell::process::CommandEvent;
            while let Some(event) = rx.recv().await {
                match event {
                    CommandEvent::Stdout(line) | CommandEvent::Stderr(line) => {
                        let text = String::from_utf8_lossy(&line);
                        if let Some(evt) = parse_yt_dlp_line(&text) {
                            let _ = on_event.send(evt);
                        }
                    }
                    CommandEvent::Terminated(status) => {
                        if status.code.unwrap_or(-1) != 0 {
                            let _ = on_event.send(DownloadEvent::Error {
                                message: format!("yt-dlp exited with code {}", status.code.unwrap_or(-1)),
                            });
                        }
                        // Clear active child
                        // Note: child has already terminated, just clean state
                        break;
                    }
                    _ => {}
                }
            }
        });

        Ok(())
    }
    ```

    Implement `update_ytdlp` command:
    ```rust
    #[tauri::command]
    pub async fn update_ytdlp(app: tauri::AppHandle) -> Result<String, String> {
        // 1. Determine current platform asset URL:
        //    - cfg!(target_os = "macos"): "https://github.com/yt-dlp/yt-dlp/releases/latest/download/yt-dlp_macos"
        //    - cfg!(target_os = "windows"): "https://github.com/yt-dlp/yt-dlp/releases/latest/download/yt-dlp.exe"
        //    - cfg!(target_os = "linux"): "https://github.com/yt-dlp/yt-dlp/releases/latest/download/yt-dlp"

        // 2. Resolve current sidecar path using the same pattern as download.rs:
        //    app.path().resolve("binaries/yt-dlp", BaseDirectory::Resource)
        //    But for update, we need to write to the ACTUAL binary location.
        //    In dev mode this is src-tauri/binaries/yt-dlp-<triple>.
        //    In release mode this is inside the app bundle resource dir.

        // 3. Download to a temp path: <sidecar_path>.new
        //    Use reqwest to download the binary bytes.
        //    Write to <sidecar_path>.new via tokio::fs::write.

        // 4. Set executable permission on macOS/Linux:
        //    #[cfg(unix)] { std::fs::set_permissions(&temp_path, std::fs::Permissions::from_mode(0o755)); }

        // 5. Atomic replace: std::fs::rename(<sidecar_path>.new, <sidecar_path>)
        //    This is safe because the binary is not currently running.

        // 6. Verify by running yt-dlp --version again and returning new version string.
    }
    ```

    Important implementation notes:
    - Use `reqwest::Client::new()` with `.header("User-Agent", "youtube-dl-app")` for GitHub API calls (GitHub rejects requests without User-Agent).
    - Write to temp path first, then `fs::rename` (atomic replace). NEVER write directly over the binary.
    - On Unix (macOS/Linux), use `std::os::unix::fs::PermissionsExt` to set mode 0o755 after download.
    - The sidecar path resolution for writing is different from reading. For the update case, resolve the full path including the target triple suffix. Use `app.shell().sidecar("binaries/yt-dlp")` to build the command (which resolves the path), but for file operations you need the actual filesystem path. Store this as a known path derived from the resource directory + target triple.
